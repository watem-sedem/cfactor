import os

from cfactor.decorators import check_nan


@check_nan
def celc_to_fahr(C):
    """
    Computes degree Celsisus

    Parameters
    ----------
        'C' (float, or pd series, array): temperature in degree celsius

    Returns
    -------
        'F'(float, or pd series, array): temperature in degree fahrenheight
    """
    return 9 / 5 * C + 32
